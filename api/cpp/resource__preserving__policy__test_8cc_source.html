27 // Test that the first ready and non-aspired version is unloaded first.
28 TEST(ResourcePreservingPolicyTest, UnloadsFirstNonAspired) {
29  std::vector<AspiredServableStateSnapshot> versions;
30  versions.push_back({{"test", 1}, LoaderHarness::State::kUnloading, false});
31  versions.push_back({{"test", 2}, LoaderHarness::State::kReady, true});
32  versions.push_back({{"test", 3}, LoaderHarness::State::kReady, false});
33  versions.push_back({{"test", 4}, LoaderHarness::State::kNew, true});
34  versions.push_back({{"test", 5}, LoaderHarness::State::kReady, false});
35 
36  ResourcePreservingPolicy policy;
37  const auto action = policy.GetNextAction(versions);
38  ASSERT_TRUE(action);
39  EXPECT_EQ(AspiredVersionPolicy::Action::kUnload, action->action);
40  EXPECT_EQ(3, action->id.version);
41 }
